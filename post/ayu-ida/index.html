A*与IDA*

2021-05-16 9 min read # 搜索 # A* # IDA*

A∗A^*A∗和IDA∗IDA^*IDA∗是一种搜索算法

A∗A^*A∗

首先不得不提优先队列的BFS算法：
就是每次都取出优先队列（二叉堆）顶部的状态，用这个状态去拓展然后得到最优解的算法
如果我们构造一个函数h(state)h(state)h(state)指当前状态与最终状态的估计代价，以及g(state)g(state)g(state)指以走过的状态设估价函数f(state)=g(state)+h(state)f(state)=g(state)+h(state)f(state)=g(state)+h(state)，将估价函数作为关键字扔入堆中，每次去除估价函数托展
不难得出该估价一定是小于等于于实际的
因为如果估价函数如果大于实际函数话，那么最优解就会因估价过大而不被取出，程序就一直在非最优的道路上拓展，最终的答案就是非最优答案。
当f=0f=0f=0时，A∗A^*A∗相当于普通的BFS，估价函数设计的优劣，关乎程序的快慢
A∗A^*A∗的代码大致为:

八数码问题

定义估价函数为当前状态与目标状态差异的个数

#include<bits/stdc++.h>
#define ll long long
#define rgi register int
#pragma comment(linker, "/STACK:16777216")
using namespace std;
int dx[4]={1,-1,0,0},dy[4]={0,0,1,-1};
inline ll read() {
	ll s=1,a=0;
	char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') s=-s;
		c=getchar();
	}
	while(isdigit(c)) {
		a=a*10+c-'0';
		c=getchar();
	}
	return s*a;
}
struct node {
	int a[5][5];
	bool operator < (node x) const {
		for(int i=1; i<=3; i++) {
			for(int j=1; j<=3; j++) {
				return a[i][j]<x.a[i][j];
			}
		}
		return false;
	}
} f,g;
set <int> b;
int h(node x) {
	int res=0;
	for(int i=1; i<=3; i++) {
		for(int j=1; j<=3; j++) {
			if(x.a[i][j]!=g.a[i][j]) res++;
		}
	}
	return res;
}
int ee(node x) {
	int res=0;
	for(int i=1;i<=3;i++) {
		for(int j=1;j<=3;j++) {
			res=res*10+x.a[i][j];
		}
	}
	return res;
}
struct data {
	node sp;
	int t;
	bool operator < (data x) const {
		return h(sp)+t>h(x.sp)+x.t;
	}
};
bool ch(int x) {
	return x>=1&&x<=3;
}
void bfs() {
	priority_queue <data> q;
	q.push({f,0});
	while(q.size()) {
		data be=q.top();
		q.pop();
		if(!h(be.sp)) {
			printf("%d\n",be.t);
			return;
		}
		int sx,sy;
		for(int i=1;i<=3;i++) {
			for(int j=1;j<=3;j++) {
				if(!be.sp.a[i][j]) sx=i,sy=j;
			}
		}
		for(int i=0;i<4;i++) {
			int rx=sx+dx[i],ry=sy+dy[i];
			if(ch(rx)&&ch(ry)) {
				swap(be.sp.a[sx][sy],be.sp.a[rx][ry]);
				if(!b.count(ee(be.sp))) b.insert(ee(be.sp)),q.push({be.sp,be.t+1});
				swap(be.sp.a[sx][sy],be.sp.a[rx][ry]);
			}
		}
	}
}
int main() {
	for(int i=1;i<=3;i++) {
		for(int j=1;j<=3;j++) {
			char c;
			cin>>c;
			f.a[i][j]=c-'0';
		}
	}
	g.a[1][1]=1,g.a[1][2]=2,g.a[1][3]=3,g.a[2][1]=8,g.a[2][2]=0,g.a[2][3]=4,g.a[3][1]=7,g.a[3][2]=6,g.a[3][3]=5;
	bfs();
	return 0;
}

骑士精神

这个和八数码问题差不多

#include<bits/stdc++.h>
using namespace std;
inline int read() {
	int s=1,a=0;
	char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') s=-s;
		c=getchar();
	}
	while(isdigit(c)) {
		a=a*10+c-'0';
		c=getchar();
	}
	return s*a;
}
int dx[8]={1,-1,2,-2,1,-1,2,-2};
int dy[8]={2,2,1,1,-2,-2,-1,-1};
struct node {
	int a[7][7];
	void init() {
		memset(a,0,sizeof(a));
	}
} f,g;
int h() {
	int res=0;
	for(int i=1;i<=5;i++) {
		for(int j=1;j<=5;j++) {
			if(f.a[i][j]!=g.a[i][j]) res++;
		}
	}
	return res;
}
bool flg;
bool ch(int x) {
	return x>=1&&x<=5;
}
void dfs(int dep,int x,int y,int maxdep) {
	if(flg) return;
	if(dep+h()>maxdep) return;
	if(!h()) {
		printf("%d\n",dep);
		flg=1;
		return;
	}
	for(int i=0;i<8;i++) {
		int sx=x+dx[i],sy=y+dy[i];
		if(ch(sx)&&ch(sy)) {
			swap(f.a[x][y],f.a[sx][sy]);
			dfs(dep+1,sx,sy,maxdep);
			swap(f.a[x][y],f.a[sx][sy]);
		}
	}
}
int main() {
	int test;
	test=read();
	for(int i=1;i<=5;i++) {
		for(int j=1;j<=5;j++) {
			if((i<3&&j>=i)||(i>=3&&j>i)) g.a[i][j]=1;
			else if(i==3&&j==3) g.a[i][j]=2;
			else g.a[i][j]=0;
		}
	}
	while(test--) {
		f.init();
		int ddx,ddy;
		for(int i=1;i<=5;i++) {
			for(int j=1;j<=5;j++) {
				char ccc;
				cin>>ccc;
				if(ccc=='*') f.a[i][j]=2,ddx=i,ddy=j;
				else f.a[i][j]=ccc-'0';
			}
		}
		if(!h()) {
			puts("0");
			continue;
		}
		flg=0;
		for(int i=1;i<=16&&!flg;i++) {
			dfs(0,ddx,ddy,i);
		}
		if(!flg) puts("-1");
	}
	return 0;
}

The Rotation Game

定义估价函数为当前中心八块与全部以当前中心出现次数最多的数字的差异
枚举每个操作转移，回溯时要进行相反的操作

#include<bits/stdc++.h>
#define ll long long
#define rgi register int
#pragma comment(linker, "/STACK:16777216")
using namespace std;
const int N=9;
inline ll read() {
	ll s=1,a=0;
	char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') s=-s;
		c=getchar();
	}
	while(isdigit(c)) {
		a=a*10+c-'0';
		c=getchar();
	}
	return s*a;
}
char ccc[31];
int mp[N*N];
int po[8][7]= {
	{0,2,6,11,15,20,22},
	{1,3,8,12,17,21,23},
	{10,9,8,7,6,5,4},
	{19,18,17,16,15,14,13},
	{23,21,17,12,8,3,1},
	{22,20,15,11,6,2,0},
	{13,14,15,16,17,18,19},
	{4,5,6,7,8,9,10}
};
int cen[9]= {6,7,8,11,12,15,16,17};
int h() {
	int cnt[3]= {0,0,0},maxchu=0;
	for(int i=0; i<8; i++) {
		cnt[mp[cen[i]]-1]++;
		maxchu=max(maxchu,cnt[mp[cen[i]]-1]);
	}
	return 8-maxchu;
}
int duiyin[9]= {5,4,7,6,1,0,3,2,-1};
void change(int x) {
	int kk=mp[po[x][0]];
	for(int i=0; i<6; i++) {
		mp[po[x][i]]=mp[po[x][i+1]];
	}
	mp[po[x][6]]=kk;
}
bool flg;
void dfs(int dep,int maxdep,int opt) {
	if(flg) return;
	if(dep>maxdep||dep+h()>maxdep) return;
//	cout<<dep<<endl;
	if(!h()) {
		flg=1;
		for(int i=0; i<dep; i++) {
			cout<<ccc[i];
		}
		cout<<endl;
		cout<<mp[cen[0]]<<endl;
		return;
	}
	for(int i=0; i<8; i++) {
		if(duiyin[opt]!=i) {
			change(i);
			ccc[dep]=i+'A';
			dfs(dep+1,maxdep,i);
			change(duiyin[i]);
		}
	}
}
int main() {
	while(true) {
		int ssh=read();
		if(ssh==0) break;
		mp[0]=ssh;
		for(int i=1; i<24; i++) {
			mp[i]=read();
		}
		if(!h()) {
			puts("No moves needed");
			printf("%d\n",mp[cen[0]]);
			continue;
		}
		flg=0;
		for(int mm=1; !flg; mm++) {
			dfs(0,mm,8);
		}
	}
	return 0;
}
